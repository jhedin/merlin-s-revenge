[#name: "ListExtractByProp(listOfPropLists, propToGet)", #scriptType: #movie, #type: #script]
on ListExtractByProp(listOfPropLists, propToGet)
  theList = []
  repeat with prList in listOfPropLists
    theList.append(prList[propToGet])
  end repeat
  return theList
end
