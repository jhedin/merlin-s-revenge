[#name: "SpriteSetMember(spr, mem, type)", #scriptType: #movie, #type: #script]
on SpriteSetMember(spr, mem, type, keepSize)
  if type = void then
    type = #gameChar
  end if
  
  if keepSize = void then
    keepSize = false
  end if
  
  case type of
    #gameChar:
      spr.member = mem
      
      if keepSize = false then
        spr.width = mem.width
        spr.height = mem.height
      end if
      
      -- if the member is black and white
      if spr.member.type = #bitmap then
        if spr.member.depth = 1 then
          spr.ink = 1 -- transparent ink, otherwise blend will not work
        else
          spr.ink = 36 -- background transparent ink for other colour depths
        end if
      else
        spr.ink = 36
      end if
      
      
    #field:
      spr.member = mem
      spr.ink = 0
  end case
end
