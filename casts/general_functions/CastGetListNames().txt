[#name: "CastGetListNames(theCast)", #scriptType: #movie, #type: #script]
on CastGetListNames(theCast)
  theList = []
  numMem = the number of members of castLib(theCast)
  repeat with mem = 1 to numMem
    nextMem = member(mem, theCast)
    theList[mem] = nextMem.name
  end repeat
  
  Return theList
end
