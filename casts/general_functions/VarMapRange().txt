[#name: "VarMapRange(var, varRange, valRange)", #scriptType: #movie, #type: #script]
on varMapRange var, varRange, valRange
  -- maps point var in varRange to the equivelent in valRange
  
  percent = VarPercent(var, varRange)
  val = VarValRange(percent, valRange)
  
  return val
end
