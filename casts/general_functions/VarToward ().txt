[#name: "VarToward (var, targit, amount)", #scriptType: #movie, #type: #script]
on VarToward (var, targit, amount)
  -- move variable <var> toward targit by <amount>
  
  if var > targit then 
    var = var - amount
    if var < targit then var = targit
  end if
  
  if var < targit then
    var = var + amount
    if var > targit then var = targit
  end if
  
  return var
end VarTowardZero
