[#name: "VarPercent (val,listRange)", #scriptType: #movie, #type: #script]
on VarPercent (val,lRange)
  -- returns a value in a range as a percentage  
  
  if val < min(lRange[1], lRange[2]) then
    return 0
  else if val > max(lRange[1], lRange[2]) then
    return 100
  end if
  
  rangelength = VarDiff(lrange[1],lrange[2])
  strange = lrange[1] * 1.000
  rangeDir = VarMoreLess(lrange[1],lrange[2])
  
  valpos = (val - strange) *1.000
  
  valpercent = (valpos/rangelength *100) * rangedir
  
  return valpercent
  
end VarPercent
