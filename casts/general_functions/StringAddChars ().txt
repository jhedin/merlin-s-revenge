[#name: "StringAddChars (thestring, numofchars, addtostart, theChar)", #scriptType: #movie, #type: #script]
on StringAddChars thestring, numofchars, addtostart, theChar
  -- makes <thestring> up to lenth <numof chars> by adding spaces
  if theChar = void then theChar = " "  
  
  thelenth = thestring.chars.count
  
  charstoadd = numofchars - thelenth
  
  if charstoadd > 0 then
    repeat with ch = 1 to charstoadd
      if addtostart then
        thestring = theChar & thestring
      else
        thestring = thestring & theChar
      end if
    end repeat
  end if
  
  return thestring
  
end StringAddChars
