[#name: "GeomHyp (a,b)", #scriptType: #movie, #type: #script]
on GeomHyp (a,b)
  -- calculates calculates the length of side hyp c
  -- right angled triangle a b c
  
  hypsqrd = (a * a) + (b*b)
  
  hyp = sqrt(hypsqrd)
  return hyp
  
end GeomHyp
