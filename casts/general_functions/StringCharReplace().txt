[#name: "StringCharReplace(str, chr, replacement)", #scriptType: #movie, #type: #script]
on StringCharReplace(theString, theChar, replacement)
  if replacement = "" then
    return StringEliminateChars(theString, theChar)
  end if
  
  strLen = theString.chars.count
  theChar = theChar.char[1]
  repeat with i = 1 to strLen
    nChar = theString.char[i]
    if nChar = theChar then
      theString = theString.char[1 .. (i-1)] & replacement & theString.char[(i+1) .. 99]
    end if
  end repeat
  
  return theString
end
