[#name: "GeomDistSqr (loc1,loc2)", #scriptType: #movie, #type: #script]

on GeomDistSqr (loc1,loc2)
  
  -- calculates the dot product of two points (aka the distance squared)
  
  
  --signs dont matter, they will be squared out
  --diffx = VarDiff (loc1[1],loc2[1])
  --diffy = VarDiff (loc1[2],loc2[2])
  distXY = loc2-loc1
  
return distXY.loch*distXY.loch + distXY.locv*distXY.locv
  
end GeomDistSqr
