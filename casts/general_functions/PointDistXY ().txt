[#name: "PointDistXY (fromLoc, toLoc)", #scriptType: #movie, #type: #script]
-- gives the x and y distances between two points
on PointDistXY (fromLoc, toLoc)
  distXY = point(0,0)
  distXY.loch = abs(fromLoc.loch - toLoc.loch)
  distXY.locv = abs(fromLoc.locv-toLoc.locv)
  
  return distXY
end
