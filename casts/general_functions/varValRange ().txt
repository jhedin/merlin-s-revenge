[#name: "varValRange (percent,listRange)", #scriptType: #movie, #type: #script]
on VarValRange (perc,lRange)
  -- returns a percentage as a value in a range  
  if perc <= 0 then
    return lRange[1]
  else if perc >= 100 then
    return lRange[2]
  end if
  
  rangelength = VarDiff(lrange[1],lrange[2])
  rangestart = lrange[1]
  rangedir = VarMoreLess(lrange[1],lrange[2])
  
  perc = perc * 1.000
  
  val = perc / 100 * rangelength
  
  val = (val*rangedir) + rangestart
  
  return val
  
end VarValRange
