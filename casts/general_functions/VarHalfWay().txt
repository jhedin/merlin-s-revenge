[#name: "VarHalfWay(var1, var2)", #scriptType: #movie, #type: #script]
on VarHalfWay var1, var2
  -- calculate the value which is half way between the other values
  
  diff = VarDiff(var1, var2)
  
  halfWay = max(var1, var2) - (diff /2)
  
  return halfWay
end
