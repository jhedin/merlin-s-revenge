[#name: "GeomPixelDist (loc1,loc2)", #scriptType: #movie, #type: #script]
on GeomPixelDist (loc1,loc2)
  -- calculates the distance between two points
  
  diffx = VarDiff (loc1[1],loc2[1])
  diffy = VarDiff (loc1[2],loc2[2])
  
  dist = max(diffx, diffy)
  
  return dist
  
end GeomPixelDist
