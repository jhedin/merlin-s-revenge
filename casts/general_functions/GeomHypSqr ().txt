[#name: "GeomHypSqr (a,b)", #scriptType: #movie, #type: #script]

on GeomHypSqr (a,b)

  -- calculates calculates the square of the length of side hyp c

  -- right angled triangle a b c

  

  hypsqrd = (a * a) + (b*b)

  

  return hypsqrd

  

end GeomHypSqr
