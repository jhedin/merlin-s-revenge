[#name: "PointArrivedNearestToTarget(theLoc, targetLoc, moveVect)", #scriptType: #movie, #type: #script]
-- determines whether a point has arrived at it's target
-- or at the nearest point to it
on PointArrivedNearestToTarget(theLoc, targetLoc, lastDistXY)
  targetReached = false
  
  distXY = PointDistXY(theLoc, targetLoc)
  if distXY.loch > lastDistXY.loch or distXY.locv > lastDistXY.locv then
    targetReached = true
  end if
  
  -- update lastDistXY
  -- object must store this for next time
  lastDistXY.loch = distXY.loch
  lastDistXY.locv = distXY.locv
  
  
  return targetReached
end
