[#name: "VarMoreLess (firstvar,secondvar)", #scriptType: #movie, #type: #script]
on VarMoreLess firV,secV
  -- result is whether secV is more(1),less(-1) or equal(0) to firV
  
  if firV > secV then res = -1
  if firV < secV then res = 1
  if firV = secV then res =0
  return res
  
end VarMoreLess
