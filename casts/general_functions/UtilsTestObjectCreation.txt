[#name: "UtilsTestObjectCreation", #scriptType: #movie, #type: #script]
global g

-- results

on UtilsTestObjectCreation me
  
  qty = 100
  
  
  pm = g.profileMaster
  pm.t("Test Object Creation qty=" & qty)
  
  
  objs = []
  
  pm.s(#create)
  -- create energyBlasts
  repeat with i = 1 to qty
    params = g.actorMaster.getParams(#newActor)
    params.typ = #spell
    params.startLoc = point(100,100)
    
    nObj = g.actorMaster.newActor(params)
    objs.append(nObj)
  end repeat
  
  pm.f()
  pm.s(#finish)
  
  -- finish them
  repeat with nObj in objs
    nObj.finish()
  end repeat
  
  pm.f()
  
  objs = []
  
  pm.s(#recreate)
  
  -- recreate them
  repeat with i = 1 to qty
    params = g.actorMaster.getParams(#newActor)
    params.typ = #spell
    params.startLoc = point(100,100)
    
    nObj = g.actorMaster.newActor(params)
    objs.append(nObj)
  end repeat
  
  pm.f()
  pm.s(#finishSecond)
  
  -- finish them
  repeat with nObj in objs
    nObj.finish()
  end repeat
  
  pm.f()
  
  
  pm.w()
  
end
