[#name: "GeomDist (loc1,loc2)", #scriptType: #movie, #type: #script]
on GeomDist (loc1,loc2)
  -- calculates the distance between two points
  -- requires GeomHyp
  
  diffx = VarDiff (loc1[1],loc2[1])
  diffy = VarDiff (loc1[2],loc2[2])
  
  dist = GeomHyp(diffx,diffy)
  
  return dist
  
end GeomDist
