[#name: "CastGetMembersContaining(phrase, theCast)", #scriptType: #movie, #type: #script]
on CastGetMembersContaining(phrase, theCast)
  memList = []
  numMembers = the number of members of castLib(theCast)
  repeat with i = 1 to numMembers
    nMem = member(i, theCast)
    if nMem.name contains phrase then
      memList.append(nMem)
    end if
  end repeat
  
  return memList
end
