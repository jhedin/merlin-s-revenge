[#name: "CounterReset (theC)", #scriptType: #movie, #type: #script]
on CounterReset (theC, whichEnd)
  
  if whichEnd = void then
    
    dir = VarMoreLess(0, theC.inc)
    
    case dir of
      1:
        whichEnd = 1
      -1:
        whichEnd = 2
    end case
    
  end if
  
  theC.theCount = theC.tim[whichEnd]
  
  theC.fin = false
  theC.looped = false
  
  Return theC
end CounterReset
