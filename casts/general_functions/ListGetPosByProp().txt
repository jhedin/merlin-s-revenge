[#name: "ListGetPosByProp(theList,  theValueSym, theValue)", #scriptType: #movie, #type: #script]
on ListGetPosByProp(theList, theValueSym, theValue)
  -- return the position in the list of prop lists theValueSym: theValue
  
  i = 0
  pos = 0
  repeat with nPropList in theList
    i = i + 1
    
    if nPropList[theValueSym] = theValue then
      pos = i
      exit repeat
    end if
    
  end repeat
  
  return pos
end
