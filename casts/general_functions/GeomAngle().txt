[#name: "GeomAngle(slope)", #scriptType: #movie, #type: #script]
on GeomAngle slope
  deltaH = slope[1]
  deltaV = slope[2]
  if deltaH <> 0 then
    slope = float (deltaV) / deltaH 
    angle = atan (slope)
    if deltaH  < 0 then
      angle = angle + pi         
    end if
  else if deltaV > 0 then
    angle = pi / 2
  else if deltaV < 0 then
    angle = (3 * pi) / 2
  else
    angle = 0
  end if
  -- Convert to degrees for .rotation
  angle = (angle * 180) / pi
  
  if angle < 0 then
    angle = 360 + angle
  end if
  
  return angle
end GeomAngle
