[#name: "GeomPointOnCircle(center, radius, angle)", #scriptType: #movie, #type: #script]
on GeomPointOnCircle cent, rad, angl
  
  theLoc = point(0,0)
  
  --angle = _root.VarChangeInRange(angle, [0, 359], -90);
  
  radian = (angl/180.0000) * PI
  theLoc[1] = cent[1] + cos(radian)*rad
  theLoc[2] = cent[2] - sin(radian)*rad
  
  return theLoc
end
