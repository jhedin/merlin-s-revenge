[#name: "StringGetPos (str, chr, dir)", #scriptType: #movie, #type: #script]
on StringGetPos (str, chr, dir)
  strLenth = str.chars.count
  
  pos = 0
  
  if dir = -1 then
    c = strLenth
  else
    c = 1
    dir = 1
  end if
  
  chr = chr.char[1]
  
  repeat with ch =1 to strLenth
    
    nextChar = str.char[c]
    if charToNum(nextChar) = charToNum(chr) then
      pos = c
      exit repeat
    end if 
    
    c = c+ dir
  end repeat
  
  
  Return pos
end StringGetCharPos
