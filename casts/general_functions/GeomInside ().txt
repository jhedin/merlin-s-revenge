[#name: "GeomInside (thePoint, theRect)", #scriptType: #movie, #type: #script]
on GeomInside (thePoint, theRect)
  pointInside = false
  if VarInRange(thePoint[1], [theRect[1],theRect[3]]) and VarInRange(thePoint[2], [theRect[2],theRect[4]]) then
    pointInside = true
  end if
  
  Return pointInside
end GeomInside
