[#name: "CastMakeNameKey (theCastNum, theMemType)", #scriptType: #movie, #type: #script]
on CastMakeNameKey (theCastNum, memberType)
  theKey = [:]
  numMem = the number of members of castLib(theCastNum)
  repeat with mem = 1 to numMem
    nextMem = member(mem, theCastNum)
    if nextMem.type = memberType then
      memSym = symbol(nextMem.name)
      theKey[memSym] = nextMem
    end if
  end repeat
  Return theKey
end CastMakeNameKey
