[#name: "RectCalcMiddle(theRect)", #scriptType: #movie, #type: #script]
on RectCalcMiddle(theRect)
  theLoc = point(0,0)
  theLoc.locH = (theRect.left + theRect.right) /2
  theLoc.locV = (theRect.top + theRect.bottom) /2
  
  return theLoc
end
