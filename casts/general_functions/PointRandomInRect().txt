[#name: "PointRandomInRect(theRect)", #scriptType: #movie, #type: #script]
-- returns a random point within a rect
on PointRandomInRect theRect
  theLoc = point(0,0)
  
  theLoc.loch = random(theRect.width) + theRect.left
  theLoc.locv = random(theRect.height) + theRect.top
  
  return theLoc
end
