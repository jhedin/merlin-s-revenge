[#name: "URLOfPage(page)", #scriptType: #movie, #type: #script]
on URLOfPage (thePage)
  -- makes a URL to the page depending on whether the movie is in authoring mode or not
  theURL = ""
  
  if the runmode contains "Plugin" then
    theURL = the moviePath & thePage
  else
    theURL = "http://localhost/themetalbox_08a/_website/dcr/games/" & thePage
  end if
  
  return theURL
end
