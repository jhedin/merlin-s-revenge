[#name: "objAiPlayer", #scriptType: #parent, #type: #script]
global g
global gGameView
global gGameName

property ancestor
property pUnstickSpell   -- used purely after restoreFromSave, spell is unstuck on the next frame because both spell and player must be fully restored before unstick can take place

on new me
  ancestor = new(script"objAiAttack")
  
  return me
end 

on init me, params
  ancestor.init(params)
  
  pUnstickSpell = false
end

on finish me
  ancestor.finish()
end

on initCharacterInfo me, characterPrg, spr, params
  ancestor.initCharacterInfo(characterPrg, spr, params)
  
  me.goMode(#playerControl)
end

on attackFin me, reason
  case reason of 
    #completed:
      me.goMode(#playerControl)
      me.pCharacterPrg.goMode(#walk)
      
  end case
  
  ancestor.attackFin(reason)
end

on characterModeChanged me, newCharMode
  case newCharMode of
    #dazed, #die:
      me.goMode(#dazed)
  end case
end

on getTarget me
  return #notApplicableToPlayer
end

on getTargetLoc me
  return g.mouseMaster.getMouseLoc()
end

on goMode me, newMode
  
  -- this was to avoid being able to move without cancelling script
  -- if spell was released during first second of script
  -- it introduced a new bug whereby game would crash if Merlin was in release mode when script
  -- started
  
  -- if we are in #freeze mode, do not change mode
  -- for #release
  
  --  if me.getMode() = #freeze and newMode = #release then
  --    return
  --  end if
  
  -- when coming out of #freeze mode, unstick the current spell
  if me.getMode() = #freeze then
    pUnstickSpell = true -- attempt to unstick the spell next frame
  end if
  
  ancestor.goMode(newMode)
end 

on internalEvent me, theEvent
  case theEvent of
    #buildingFinished:
      me.goMode(#playerControl)
    #spellCharged:
      if me.pCharacterPrg.getGmgOn() then
        if me.getAttack().gmgAutoFire then
          me.playerAttackRelease()
          me.playerAttackCharge()
        end if
      end if
    #gmgTurnedOn:
      me.playerAttackRelease()
    #gmgTurnedOff:
      me.playerAttackRelease()
      me.playerAttackCharge()
  end case
end

on interpretMouse me
  mouseState = g.mouseMaster.getMouseState()
  case gGameName of
      
    #merlin_3:
      
      if me.pMode = #freeze then
        
        case mouseState of
          #pressed, #released:
            me.pCharacterPrg.AIisTryingToMove()
        end case
        
        
      else
        
        case mouseState of
          #notPressed:
            nothing
            
          #pressed:
            me.playerAttackCharge()
            
          #released:
            me.playerAttackRelease()
            
        end case
        
        
      end if
      
  end case
end

on interpretMoveKeys me
  moveVector = g.keyMaster.getMoveVector()
  
  
  if me.pMode = #freeze then
    if moveVector <> point(0,0) then
      me.pCharacterPrg.AIisTryingToMove()
    end if
    
    moveVector = point(0,0)
  end if
  
  me.pCharacterPrg.moveHoriz(moveVector[1])
  
--  case gGameView of

--      
--    #topDown:
      me.pCharacterPrg.moveVert(moveVector[2])
      
--  end case
  
  me.recordMoveVector(moveVector)
end

on playerAttackCharge me
  -- if in dialogue mode
  if me.pMode = #freeze then
    me.pCharacterPrg.AIisTryingToMove()
    return
  end if
  
  -- #melee and #ranged will autofire
  -- #magic must be charged and released
  
  -- this implements the cooldown on spells
  me.attack()
  
  -- to avoid the cooldown comment out the above and uncomment the following:
  
  --  attackType = me.getAttackType()
  --  
  --  case attackType of
  --    #magic:
  --      me.chargeMagic()
  --      
  --    otherwise:
  --      me.attack()
  --  end case
end

on playerAttackRelease me
  attackType = me.getAttack().type
  
  case attackType of
    #melee, #ranged:
      nothing
      
    #magic:
      targetLoc = me.getTargetLoc()
      me.releaseMagic(targetLoc)
  end case
end

on restoreFromSave me, sd
  ancestor.restoreFromSave(sd)
  
  pUnstickSpell = true -- attempt to unstick spell next frame.
end

on restorePlayerControl me
  me.goMode(#playerControl)
end

on start me
  ancestor.start()
  
  me.goMode(#beBuilt)
end

on unpaws me
  ancestor.unpaws()
  
  me.unstickCurrentSpell()
end

on unstickCurrentSpell me
  -- to prevent energy blast from staying in place upon unpause
  playerMode = me.pCharacterPrg.getMode()
  
  if playerMode = #charge then
    -- without this, misses presses made during map init phase of loading saved game
    g.mouseMaster.checkMouse()
    
    mouseState = g.mouseMaster.getMouseState()
    
    if mouseState = #released or mouseState = #notPressed then
      me.playerAttackRelease()
    end if
    
  end if
end


on update me
  if pUnstickSpell = true then
    me.unstickCurrentSpell()
    pUnstickSpell = false
  end if
  
  case me.pMode of
    #attack, #playerControl, #freeze, #release:
      me.interpretMoveKeys()
      me.interpretCheatKeys()
      me.interpretGameKeys()
      me.interpretMouse()
      
  end case
  
  
  
  ancestor.update()
end
