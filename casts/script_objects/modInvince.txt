[#name: "modInvince", #scriptType: #parent, #type: #script]
-- used by any game object that needs to be invincible
-- for periods of time
-- at the time of writing that's just Merlin
-- invince is used rather than invincibility because of difficult to spot typing errors

property ancestor
property pInvinceActive       -- whether or not the character is currently invicible
property pInvinceLockOn       -- whether the invincibility is locked "on" by player pressing cheat key. this prevent switch off when timer runs out on temp invincibility
property pTempInvinceCounter  -- counter to keep track of when temporary invincibility runs out
property pTempInvinceTime     -- how long (in frames) temp invincibility lasts for

on new me
  ancestor = new (script("modModule"))
  return me
end

on addModParams me
  i = me.modifyParams(#init)
  
  ancestor.addModParams()
end

on init me, params  
  ancestor.init(params)
  
  pInvinceActive = false
  pInvinceLockOn = false
  
  pTempInvinceTime = 200
  
  pTempInvinceCounter = CounterNew()
  pTempInvinceCounter.tim[2] = pTempInvinceTime
end

on addSaveData me, sd
  ancestor.addSaveData(sd)
  
  sd[#pInvinceActive] = pInvinceActive
  sd[#pInvinceLockOn] = pInvinceLockOn
  sd[#pTempInvinceCounter] = pTempInvinceCounter
end

on getInvinceActive me
  -- if modInvincibility is not installed, modCatcher will return false
  return pInvinceActive
end

on invinceOn me
  pInvinceActive = true
  
  me.ID.bigMe.pulseWhite()
end

on invinceOff me
  pInvinceActive = false
  
  me.ID.bigMe.pulseWhiteStop()
end

on invinceToggle me
  
  if pInvinceActive then
    me.invinceOff()
    pInvinceLockOn = false
  else
    me.invinceOn()
    pInvinceLockOn = true
  end if
  
end

on restoreFromSave me, sd
  ancestor.restoreFromSave(sd)
  
  pInvinceActive = sd.pInvinceActive
  pInvinceLockOn = sd.pInvinceLockOn
  pTempInvinceCounter = sd.pTempInvinceCounter
end

on startTempInvince me
  CounterReset(pTempInvinceCounter)
  
  if pInvinceLockOn = false then
    me.invinceOn()
  end if
end

on update me
  ancestor.update()
  
  if pInvinceActive then
    me.updateTempInvince()
  end if
end

on updateTempInvince me
  if pInvinceLockOn = false then
    
    CounterOnce(pTempInvinceCounter)
    
    if pTempInvinceCounter.fin then
      me.invinceOff()
    end if
    
  end if
end
