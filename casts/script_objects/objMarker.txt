[#name: "objMarker", #scriptType: #parent, #type: #script]
global g

property ancestor
property pSpr

on new me
  ancestor = new (script "objBasic")
  me.init()
  return me
end

on init me
  pSpr = g.spriteMaster.requestSprite()
  SpriteSetMember(pSpr, member("markerBoundary", "gfx"))
end

on setRect me, theRect
  pSpr.rect = theRect
end
