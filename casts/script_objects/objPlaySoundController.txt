[#name: "objPlaySoundController", #scriptType: #parent, #type: #script]
global g

property ancestor

on new me
  ancestor = new(script"objController")
  
  i = me.modifyParams(#init)
  
  i.exclusive = false
  
  return me
end

on newObject me, definition, theLoc, caller
  definitionText = StringEliminateChars(definition.text, return)
 
  newObj = g.objectMaster.requestObject(#objPlaySound)
  params = newObj.getParams(#init)
  params.definition = definitionText
  newObj.init(params)
  
  newObj.playSound()
  
  me.pObjects.append(newObj)
  
  return newObj
end
