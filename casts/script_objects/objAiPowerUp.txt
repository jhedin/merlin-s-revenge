[#name: "objAiPowerUp", #scriptType: #parent, #type: #script]
global g

property ancestor
property pPlayer

on new me
  ancestor = new(script"objAi")
  return me
end 

on init me, params
  ancestor.init(params)
  
  pPlayer = g.actorMaster.getPlayer()
  me.goMode(#norm)
end

on refreshPlayer me
  if pPlayer = #none then
    pPlayer = g.actorMaster.getPlayer()
  end if
end


on update me
  case me.pMode of
    #norm:
      if me.pCharacterPrg.checkForCollisionWithPlayer() then
        me.refreshPlayer()
        me.pCharacterPrg.collected(pPlayer)
        me.goMode(#dead)
      end if
      
  end case
end
