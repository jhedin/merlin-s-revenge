[#name: "modModule", #scriptType: #parent, #type: #script]
-- these object's ancestors are set dynamically to allow them to be
-- part of a chain

property ancestor

on new me
  return me
end

on addModParams me
  -- this should be overridden with object specific version to add params
  -- remember to pass it on, otherwise later modules will not add their params
  
  ancestor.addModParams()
end

on setAncestor me, newVal
  ancestor = newVal
end
