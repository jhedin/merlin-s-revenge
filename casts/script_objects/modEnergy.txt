[#name: "modEnergy", #scriptType: #parent, #type: #script]
-- used by any game object that needs to have
-- health

property ancestor
property pDamageSpeed
property pEnergy
property pEnergyIncAmount       -- the amount to increase the energy by each level up (50% of original level)
property pEnergyRecoverCounter
property pGlowREdPercentage     -- the percentage at which to glow red
property pKilledInAction        -- whether the unit went down fighting (if not, it was just cleared from the screen by some other process)
property pMaxEnergy             -- the max no of hitpoints it's possible for this unit to have
property pMinEnergy             -- the amount of energy below which the unit will die (usually used for mulistage enemies)
property pTakeHitSound          -- the sound to make when the object gets hit
property pTakeHitVolume    -- the volume to play the takeHitSound at

on new me
  ancestor = new (script("modModule"))
  return me
end

on addModParams me
  i = me.modifyParams(#init)
  
  i[#damageSpeed] = 5
  i[#energy] = 100
  i[#energyIncPercentage] = 1
  i[#energyRecoverDelay] = 1000
  i[#maxEnergy] = #auto
  i[#minEnergy] = 0
  i[#takeHitSound] = #none
  i[#takeHitVolume] = #none
  
  ancestor.addModParams()
end

on init me, params
  
  pDamageSpeed = params.damageSpeed
  pEnergy = params.energy
  pEnergyRecoverCounter = CounterNew()
  pEnergyRecoverCounter.tim[2] = params.energyRecoverDelay
  pGlowRedPercentage = 50
  pKilledInAction = false
  if params.maxEnergy = #auto then
    pMaxEnergy = pEnergy
  else
    pMaxEnergy = params.maxEnergy
  end if
  pMinEnergy = params.minEnergy
  pTakeHitSound = params.takeHitSound
  pTakeHitVolume = params.takeHitVolume
  
  -- energy inc amount = 5% of original energy
  pEnergyIncAmount = params.energy * params.energyIncPercentage / 100
  
  ancestor.init(params)
end

on addSaveData me, sd
  ancestor.addSaveData(sd)
  
  sd[#pEnergy] = pEnergy
  sd[#pMaxEnergy] = pMaxEnergy
end

on addToEnergy me, amount
  me.increaseEnergy(amount)
end

on calcEnergyRectBottom me
  -- returns a rect used to position an energy bar underneath us
  standMember = me.getAnimMemberFromStripAt(#stand, 1)
  myLoc = me.getLoc().duplicate()
  reg = standMember.regPoint.duplicate()
  energyRect = standMember.rect + rect(myLoc.locH, myLoc.locV, myLoc.locH, myLoc.locV) - rect(reg.locH, reg.locV, reg.locH, reg.locV)
  return energyRect
end

on checkDead me
  if pEnergy <= pMinEnergy then
    return true
  end if
  return false
end

on checkEnergyIsAtMax me
  if pMaxEnergy > pEnergy then
    return false
  end if
  
  return true
end

on die me
  ancestor.die()
  
  me.loseAllEnergy()
end

on energyChanged me
  -- override to update energy bar
end

on getEnergy me
  return pEnergy
end

on getHealth me
  -- returns the object's energy as a percentage
  
  health = VarPercent(pEnergy, [0, pMaxEnergy])
  
  return health
end

on getKilledInAction me
  return pKilledInAction
end

on getMaxEnergy me
  return pMaxEnergy
end

on glowRedOnLowHealth me
  health = me.getHealth()
  
  if health < pGlowRedPercentage then
    me.big.glowRed()
  end if
end

on increaseEnergy me, amount
  pEnergy = pEnergy + amount
  
  if pEnergy > pMaxEnergy then
    pEnergy = pMaxEnergy
  end if
  
  -- check whether to stop glowing red
  health = me.getHealth()
  if health >= pGlowRedPercentage then
    me.big.stopGlowRed()
  end if
  
  me.ID.bigMe.energyChanged()
end

on increaseEnergyByPercentage me, percent
  amount = VarValRange(percent, [0, pMaxEnergy])
  
  me.increaseEnergy(amount)
end

on internalEvent me, theEvent
  ancestor.internalEvent(theEvent)
  
  case theEvent of
    #colourTransformFin:
      me.glowRedOnLowHealth()
      
    #levelUp:
      me.levelUpEnergy()
      
  end case
  
end

on levelUpEnergy me
  --  -- increase energy by 20%
  --  incAmount = pMaxEnergy * 20 / 100
  
  incAmount = pEnergyIncAmount
  pMaxEnergy = pMaxEnergy + incAmount
  
  -- add this much onto current energy
  -- otherwise it looks as if the unit has lost energy
  me.increaseEnergy(incAmount)
  
  me.big.internalEvent(#maxEnergyChanged)
end

on loseAllEnergy me
  me.loseEnergy(pMaxEnergy)
end

on loseEnergy me, amount
  -- only lose energy if the character is not invincible
  if me.ID.bigMe.getInvinceActive() = false then
    
    pEnergy = pEnergy - amount
    me.ID.bigMe.energyChanged()
    
    if me.checkDead() then
      me.ID.bigMe.outOfEnergy()
      me.big.internalEvent(#outOfEnergy)
      me.eventNotify(#outOfEnergy)
      -- make sure any later recoverEnergy calls don't recover us into living
      pEnergy = -100
      -- record that we were killed during gameplay
      pKilledInAction = true
    else
      me.ID.bigMe.flickWhite()
    end if
    
    me.big.playSound(pTakeHitSound, pTakeHitVolume)
    
  end if
end

on outOfEnergy me
  -- override this to respond to being out of energy
  ancestor.outOfEnergy()
end

on recoverEnergy me
  if me.ID.bigMe.checkDead() = false then
    if pEnergy < pMaxEnergy then
      if pEnergyRecoverCounter.fin then
        
        me.increaseEnergy(1)
      end if
      
      Counter(pEnergyRecoverCounter)
    end if
  end if
end

on restoreFromSave me, sd
  ancestor.restoreFromSave(sd)
  
  pEnergy = sd.pEnergy
  pMaxEnergy = sd.pMaxEnergy
  me.big.energyChanged(#restoreFromSave)
  me.big.internalEvent(#maxEnergyChanged)
end

on restoreEnergy me
  if pEnergy < pMaxEnergy then
    pEnergy = pMaxEnergy
    me.ID.bigMe.energyChanged(#restoreEnergy)
  end if
end

on setEnergy me, amount
  pEnergy = amount
end

on takeDamage me, amount
  if amount > pDamageSpeed then
    amount = amount - pDamageSpeed
    me.loseEnergy(amount)
  end if
end

on takeHeal me, collisionVect, healingObj
  collSpeedX = VarPositive(collisionVect[1])
  collSpeedY = VarPositive(collisionVect[2])
  healAmount = (collSpeedX + collSpeedY) * 2
  me.increaseEnergy(healAmount)
  
  ancestor.takeHeal(collisionVect, healingObj)
  
  me.big.glowGold()
end

on takeHit me, collisionVect, attackingObj, owner
  -- it is importantant that modExperience updates the
  -- attacking obj before we find out whether we have run out of energy 
  -- otherwise exp could go to the wrong unit
  
  if me.big.getMode() <> #recoil then
    ancestor.takeHit(collisionVect, attackingObj, owner)
    collSpeedX = VarPositive(collisionVect[1])
    collSpeedY = VarPositive(collisionVect[2])
    multiplier = attackingObj.getAttack().damageMultiplier
    damage = (collSpeedX + collSpeedY) * multiplier
    if(damage > 0) then
      me.loseEnergy(damage)
    end if
  end if
  
end
