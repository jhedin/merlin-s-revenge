[#name: "objEditField", #scriptType: #parent, #type: #script]
property ancestor
property pLayer
property pLocation

on new me
  ancestor = new (script"objSpriteMember")
  
  i = me.modifyParams(#init)
  
  i.memberType = #field
  
  i[#location] = point(10,10)
  i[#layer] = 99
  
  return me
end

on init me, params
  pLayer = params.layer
  pLocation = params.location
  
  ancestor.init(params)
end

on display me
  me.requestMember()
  me.setMember()
  me.setMemberName("inputBox")
  
  me.setSpriteLayer(pLayer)
  me.setSpriteLoc(pLocation)
end
