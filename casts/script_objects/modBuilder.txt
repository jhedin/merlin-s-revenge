[#name: "modBuilder", #scriptType: #parent, #type: #script]

-- allows the unit to build other units or buildings

-- most of the logic is handled by objAICPUBuilder



property ancestor

property pBuilding      -- the building we are building

property pBuildRate     -- the rate at which we are building (100 = the animation speed of the built object)

property pBuildRateInc  -- the increase when we level

property pBuildRange    -- the max distance away from the building we can be and still build it

property pUnitToBuild   -- the unit that this unit will build

property pBuildDie      -- Die after building

property pBuildOne      -- Build only once



on new me

  ancestor = new (script("modModule"))

  return me

end



on addModParams me

  i = me.modifyParams(#init)

  

  i[#unitToBuild] = [#none,#none, #none, #none]

  i[#buildRate] = 100

  i[#buildRateInc] = 50

  i[#buildDie] = false

  i[#buildOne] = true

  ancestor.addModParams()

end



on init me, params

  ancestor.init(params)

  

  pBuilding = #none

  pBuildRate = params.buildRate

  pBuildRateInc = params.buildRateInc

  pBuildRange = 50

  pUnitToBuild = params.unitToBuild

  pBuildDie = params.buildDie

  pBuildOne = params.buildOne

end



on addSaveData me, sd

  ancestor.addSaveData(sd)

  

  me.addToSaveData(sd)

end



on addToArmyDetails me

  ad = me.big.getArmyDetails()

  

  me.addToSaveData(ad)

end



on getBuildDie me

  return pBuildDie

end



on getBuildOne me

  return pBuildOne

end



on addToSaveData me, sd

  

  sd[#pBuildRate] = pBuildRate  

end



on alignToBuilding me

  buildingRect = pBuilding.getRect()

  

  --faceDir = me.big.getFlipAsDir()

  faceDir = 1

  

  case faceDir of

    -1:

      xLoc = me.big.positionLeftEdge(buildingRect.right)

      

    1:

      xLoc = me.big.positionRightEdge(buildingRect.left)

  end case

  

  yLoc = me.big.positionBottomEdge(buildingRect.bottom)

  pBuilding.getLocZ()

  me.big.moveTowardsLoc(point(xLoc, yLoc))

  

  -- moveTowardsLoc will flip us to face the direction we are moving

  -- this will override that to make us face the building

  me.big.faceObject(pBuilding)

  

  -- make sure we appear in front of the building

  buildingLocZ = pBuilding.getLocZ()

  me.big.setLocZ(buildingLocZ + 1)

end



on checkBuildingInRange me, theBuilding

  inRange = false

  

  -- may be called before relationships are

  -- restored

  if theBuilding <> #none then

    buildRange = me.getBuildRange()

    

    distToBuilding = GeomDistSqr(me.getLoc(), theBuilding.getLoc())

    

    if distToBuilding <= (buildRange*buildRange) then

      inRange = true

    end if

    

  end if

  

  return inRange

end



on checkMyBuildingInRange me

  return me.checkBuildingInRange(pBuilding)

end



on getBuildRate me

  return pBuildRate

end



on getBuildRange me

  return pBuildRange

end



on getUnitToBuild me

  return pUnitToBuild[random(pUnitToBuild.count)]

end



on goMode me, newMode

  

  ancestor.goMode(newMode)

end 



on incBuildRate me

  pBuildRate = pBuildRate + pBuildRateInc

end



on internalEvent me, theEvent

  ancestor.internalEvent(theEvent)

  

  case theEvent of

    #addToArmyDetails:

      me.addToArmyDetails()

      

    #levelUp:

      me.incBuildRate()

      

    #restoreFromArmyDetails:

      me.restoreFromArmyDetails()

      

      

  end case

  

end



on restoreFromSave me,sd

  ancestor.restoreFromSave(sd)

  

  me.restoreFromSaveData(sd)

end



on restoreFromArmyDetails me

  ad = me.big.getArmyDetails()

  

  me.restoreFromSaveData(ad)

end



on restoreFromSaveData me, sd

  pBuildRate = sd.pBuildRate

end



on setBuilding me, newValue

  pBuilding = newValue

end



on update me

  ancestor.update()

  

  case me.big.pMode of

    #build:

      me.alignToBuilding()

  end case

end
